// --- FUSB302 Register Definitions (Corrected from fusb302.h) ---
#define FUSB302_REG_CONTROL0    0x06
#define FUSB302_REG_CONTROL1    0x07
#define FUSB302_REG_CONTROL3    0x09
#define FUSB302_REG_STATUS1     0x41
#define FUSB302_REG_INTERRUPTA  0x3E
#define FUSB302_REG_STATUS0     0x40
#define FUSB302_REG_FIFOS       0x43

// --- Control & Status Bits (Corrected from fusb302.h) ---
#define CTL0_TX_FLUSH           FUSB302_CTL0_TX_FLUSH       // 0x06, bit 6
#define CTL0_TX_START           FUSB302_CTL0_TX_START       // 0x06, bit 0
#define CTL1_RX_FLUSH           FUSB302_CTL1_RX_FLUSH       // 0x07, bit 2
#define STATUS1_RX_EMPTY        FUSB302_STATUS1_RX_EMPTY    // 0x41, bit 5
#define INTA_RX_STAT            FUSB302_INT_CRC_CHK         // Use CRC_CHK for message received in status0 (0x40, bit 4)
#define INT_I_RXHRST            FUSB302_INTA_HARDRST        // 0x3E, bit 0
#define INT_I_TXSENT            FUSB302_INTA_TXSENT         // 0x3E, bit 2
#define INT_I_RETRYFAIL         FUSB302_INTA_RETRYFAIL      // 0x3E, bit 4
#define INT_I_RXDETECT          FUSB302_INT_COMP_CHNG       // Use COMP_CHNG for connection detection (0x42, bit 5)
#define STATUS0_COMP_CHANGE     FUSB302_STATUS0_COMP        // Check COMP status change in STATUS0
#define CTL3_SEND_HARD_RESET    FUSB302_CTL3_SEND_HARD_RESET// 0x09, bit 6


// --- PD Tokens (Corrected from fusb302.h) ---
// Note: FUSB302 handles the actual SOP (0x85) internally. 
// We write the token stream markers to the FIFO.
#define PD_SOP_TKN              FUSB302_TX_TKN_SOP1         // Use SOP1 for standard message
#define PD_EOP_TKN              FUSB302_TX_TKN_EOP
#define PD_TX_OFF_TKN           FUSB302_TX_TKN_TXOFF
#define PD_PACKSYM_TKN          FUSB302_TX_TKN_PACKSYM      // 0x80
#define PD_JAMCRC_TKN           FUSB302_TX_TKN_JAMCRC       // 0xFF

// --- PD Message Data Structures ---

typedef union {
    uint32_t val;
    struct {
        uint8_t byte[4];
    } s;
} pd_data_obj_t;

typedef struct {
    uint16_t header;         
    uint8_t  msg_type;       
    uint8_t  data_role;      
    uint8_t  spec_rev;       
    uint8_t  power_role;     // New field for clarity (Source/Sink)
    uint8_t  data_obj_count; 
    uint8_t  msg_id;         
    pd_data_obj_t data[7]; 
} pd_msg_t;

// --- Policy Engine State Machine & Global Data ---

typedef enum {
    PE_INIT,
    PE_SRC_DISCOVERY,         // Initial state to detect connection
    PE_SRC_SEND_CAPABILITIES, // Sending Source_Capabilities
    PE_SRC_WAIT_REQUEST,      // Waiting for Request
    PE_SRC_TRANSITION_SUPPLY, // VBUS ramping up/down
    PE_SRC_READY,             // Contract established
    // ... (Sink states would follow)
} pe_state_t;

static pe_state_t g_pd_state = PE_INIT;
static pd_msg_t g_rx_message;
static uint8_t g_msg_id_counter = 0; // Global message ID counter (0-7)

// --- External HAL functions based on pd_debug.c ---
// NOTE: These match the signatures of functions available in the pd_debug.c context.
extern uint8_t i2c_read_reg(uint8_t reg);
extern void i2c_write_reg(uint8_t reg, uint8_t val);
extern void fusb_read_fifo(uint8_t *data, size_t len);
extern uint32_t pd_calculate_crc(const uint8_t *data, size_t len); // CRC function from prior response

// --- Helper function to construct a basic Source_Capabilities message ---
static void build_src_capabilities_msg(pd_msg_t *msg)
{
    // Use the global counter for the new message ID
    msg->msg_id = g_msg_id_counter; 
    
    // PD Spec Revision 3.0 (0x02)
    msg->spec_rev = 0x02; 
    // Data Role is DFP (0x01)
    msg->data_role = 0x01; 
    // Power Role is Source (0x01)
    msg->power_role = 0x01; 
    
    // Message Type: Source_Capabilities (Data Message Type 1)
    msg->msg_type = 0x01; 
    
    // Assume one 5V/3A Fixed Supply PDO
    msg->data_obj_count = 1;
    
    // Fixed Supply PDO: 
    // Bits 31-30=0 (Fixed), V=5000mV (10mV unit) -> 500 = 0x1F4
    // I=3000mA (10mA unit) -> 300 = 0x12C
    msg->data[0].val = (0 << 30) | (500 << 10) | (300); 
    msg->data[0].val |= (1 << 28); // Dual-Role Power (DRP) capable
}

// =============================================================================
// REQUIRED FUNCTION 1: pd_process_rx_fifo()
// (Function body remains structurally the same, but uses corrected constants)
// =============================================================================
bool pd_process_rx_fifo(pd_msg_t *msg)
{
    // The FUSB302 FIFO contains Header (2 bytes) + Data (N*4 bytes) + CRC (4 bytes).
    // The header for the message is pre-read/decoded in the pd_debug.c example,
    // but the Policy Engine needs the full raw packet for CRC calculation.
    // To match the Policy Engine logic, we'll read the full *core* packet (Header+Data+CRC).
    
    uint8_t header_bytes[2];
    uint8_t rx_buf[30]; // Max size: Header(2) + Data(7*4=28) = 30
    uint8_t *core_data = rx_buf;
    int data_len = 0;

    // Read the Header first (since we can't reliably read variable lengths without it)
    // NOTE: The FUSB302's FIFO is strange. When reading FIFOS, the first 2 bytes are SOP tokens, 
    // the next 2 are the Header. We read those first to know the length.
    // Since the pd_debug.c example pre-reads the FIFO, let's simplify to match a typical driver flow:
    
    // 1. Read the raw header bytes (Header is usually the first 2 bytes after tokens)
    // NOTE: In the FUSB302, tokens are handled by the PHY. The FIFO contains the Header, Data, and CRC.
    // The safest way is to read the header and then use it to determine the length.
    fusb_read_fifo(header_bytes, 2); 
    msg->header = header_bytes[0] | (header_bytes[1] << 8);

    // 2. Extract Data Object Count (DOC) from the header
    msg->data_obj_count = (msg->header >> 12) & 0x07; // Bits 14:12
    
    // 3. Calculate remaining bytes to read (Data + CRC)
    data_len = (msg->data_obj_count * 4) + 4; // Data(DOC*4) + CRC(4)
    if (data_len > sizeof(rx_buf)) return false; // Max data objects exceeded

    // 4. Read remaining bytes (Data Objects and CRC)
    fusb_read_fifo(rx_buf, data_len);
    
    // 5. Prepare the data for CRC check (Header + Data)
    uint8_t crc_check_buf[2 + data_len - 4]; // Header(2) + Data(N*4)
    memcpy(crc_check_buf, header_bytes, 2);
    memcpy(crc_check_buf + 2, rx_buf, data_len - 4); // Copy Data Objects

    // 6. Check CRC (CRC is the last 4 bytes of the data read)
    uint32_t received_crc = rx_buf[data_len-4] | (rx_buf[data_len-3] << 8) |
                            (rx_buf[data_len-2] << 16) | (rx_buf[data_len-1] << 24);
    
    uint32_t calculated_crc = pd_calculate_crc(crc_check_buf, sizeof(crc_check_buf));

    // 7. Flush RX FIFO immediately (as done in pd_debug.c)
    i2c_write_reg(FUSB302_REG_CONTROL1, CTL1_RX_FLUSH);

    if (received_crc != calculated_crc) return false; 

    // 8. Final Message Parsing (extracting all header fields)
    msg->msg_type = (msg->header) & 0x1F;            // Bits 4:0
    msg->data_role = (msg->header >> 5) & 0x01;      // Bit 5
    msg->spec_rev = (msg->header >> 6) & 0x03;       // Bits 6-7
    msg->power_role = (msg->header >> 8) & 0x01;     // Bit 8
    msg->msg_id = (msg->header >> 9) & 0x07;         // Bits 9-11
    
    // 9. Copy Data Objects
    for (int i = 0; i < msg->data_obj_count; i++) {
        // Data objects start at index 0 of the rx_buf
        memcpy(msg->data[i].s.byte, &rx_buf[i * 4], 4);
    }

    return true; 
}

// =============================================================================
// REQUIRED FUNCTION 2: pd_send_message()
// (Function body remains structurally the same, but uses corrected tokens/bits)
// =============================================================================
bool pd_send_message(const pd_msg_t *msg)
{
    // 1. Flush the TX FIFO before starting
    i2c_write_reg(FUSB302_REG_CONTROL0, CTL0_TX_FLUSH);

    // 2. Prepare the packet (Tokenized Stream)
    uint8_t tx_buf[32]; 
    int index = 0;
    
    // Data to be CRC'd (Header + Data)
    uint8_t crc_data[30];
    int crc_index = 0;

    // a. Start-of-Packet Token (SOP1 for FUSB302)
    tx_buf[index++] = PD_SOP_TKN; 

    // b. PD Header (16 bits, LSB first)
    // Reconstruct the header using all fields including Power Role (bit 8)
    uint16_t header = msg->msg_type 
                      | (0x00 << 4) // Reserved
                      | (msg->data_role << 5)
                      | (msg->spec_rev << 6)
                      | (msg->power_role << 8) // Power Role (Source/Sink)
                      | (g_msg_id_counter << 9) // Use global message ID
                      | (msg->data_obj_count << 12);
    
    // Store header in CRC buffer
    crc_data[crc_index++] = (uint8_t)(header & 0xFF);     
    crc_data[crc_index++] = (uint8_t)(header >> 8);       
    
    // Write PACKSYM token + Header Length (2 bytes)
    tx_buf[index++] = PD_PACKSYM_TKN | 2; 
    tx_buf[index++] = (uint8_t)(header & 0xFF);
    tx_buf[index++] = (uint8_t)(header >> 8);
    
    // c. Data Objects (4 bytes each)
    if (msg->data_obj_count > 0) {
        // Write PACKSYM token + Data Length (DOC * 4 bytes)
        tx_buf[index++] = PD_PACKSYM_TKN | (msg->data_obj_count * 4); 
        for (int i = 0; i < msg->data_obj_count; i++) {
            // Write data to TX buffer
            memcpy(&tx_buf[index], msg->data[i].s.byte, 4);
            index += 4;
            // Copy data to CRC buffer
            memcpy(&crc_data[crc_index], msg->data[i].s.byte, 4);
            crc_index += 4;
        }
    }

    // d. Calculate 32-bit JAMCRC over Header + Data
    uint32_t crc = pd_calculate_crc(crc_data, crc_index); 
    
    // e. Write JAMCRC Token
    tx_buf[index++] = PD_JAMCRC_TKN; 

    // f. End-of-Packet Token (EOP)
    tx_buf[index++] = PD_EOP_TKN; 

    // g. TX_OFF Token (Signal end of stream)
    tx_buf[index++] = PD_TX_OFF_TKN; 

    // 3. Write the entire buffer to the FIFOS register
    // NOTE: The fusb_write_reg_nbytes helper in pd_debug.c would be best here,
    // but we'll stick to a loop to match the previous structure.
    for (int i = 0; i < index; i++) {
        i2c_write_reg(FUSB302_REG_FIFOS, tx_buf[i]);
    }

    // 4. Start transmission (write to CONTROL0 to start TX)
    i2c_write_reg(FUSB302_REG_CONTROL0, CTL0_TX_START); 
    
    // 5. Increment global message ID counter
    g_msg_id_counter = (g_msg_id_counter + 1) & 0x07;

    return true;
}

// =============================================================================
// REQUIRED FUNCTION 3: Complete the State Machine (fusb302_handle_irq)
// =============================================================================
void fusb302_handle_irq(void)
{
    // 1. Read the FUSB302 interrupt status registers 
    uint8_t interrupt_a = i2c_read_reg(FUSB302_REG_INTERRUPTA);
    uint8_t interrupt_c = i2c_read_reg(FUSB302_REG_INTERRUPT); // Use INTERRUPT (0x42) for COMP_CHNG
    uint8_t status0 = i2c_read_reg(FUSB302_REG_STATUS0);
    
    // FUSB302 interrupts are generally cleared by reading, but we must explicitly 
    // address all flags to ensure the policy engine responds correctly.
    
    // A. Handle CC Detection Change (Start of connection)
    if (interrupt_c & INT_I_RXDETECT) {
        
        // This usually signals a change in the CC pin comparator status.
        // We must check STATUS0 and SWITCHES registers to see if we transitioned 
        // from 'Open' to 'Rd' (Sink attached).
        
        if (g_pd_state == PE_INIT || g_pd_state == PE_SRC_DISCOVERY) {
            
            // Assume initial negotiation starts on COMP_CHNG if we are a Source
            // and the comparator detects an Rd (Sink). 
            // The logic here is highly specific to CC detection/role.
            
            // Simplification: Assume COMP_CHNG means a device is attached, 
            // and we should proceed to send capabilities.
            
            g_pd_state = PE_SRC_DISCOVERY; // Acknowledge attachment
            
            // Immediately transition to sending capabilities (or wait for timer)
            // For a simple Policy Engine, we trigger the capability message now.
            // Check status0 to confirm CC line is active
            if (status0 & STATUS0_COMP_CHANGE) { 
                
                // Transition to the capability sending state.
                pd_msg_t cap_msg;
                build_src_capabilities_msg(&cap_msg); 
                
                if (pd_send_message(&cap_msg)) {
                    g_pd_state = PE_SRC_WAIT_REQUEST; 
                }
            }
        }
    }

    // B. Handle RX Status/Message Received (CRC_CHK in INTERRUPT (0x42))
    if (interrupt_c & FUSB302_INT_CRC_CHK) {
        
        // FUSB302 automatically sends GoodCRC, the interrupt means a valid message 
        // (Header + Data + CRC) has been received and is in the FIFO.
        
        if (pd_process_rx_fifo(&g_rx_message)) {
            
            uint8_t rx_msg_type = g_rx_message.msg_type;

            switch (g_pd_state) {
                
                case PE_SRC_WAIT_REQUEST:
                    // Waiting for Request (Data Message Type 2)
                    if (rx_msg_type == 0x02) { 
                        uint8_t response_id = g_rx_message.msg_id;
                        
                        // TODO: Implement Request validation logic here.
                        bool request_is_valid = true; 

                        if (request_is_valid) { 
                            // Send ACCEPT (Control Message Type 3)
                            pd_msg_t accept_msg = { 
                                .msg_type = 0x03, 
                                .data_obj_count = 0, 
                                .msg_id = response_id, // Match received ID
                                .spec_rev = 0x02,
                                .data_role = 0x01,
                                .power_role = 0x01
                            };
                            pd_send_message(&accept_msg);
                            
                            // NOTE: VBUS change logic needed here.
                            
                            g_pd_state = PE_SRC_TRANSITION_SUPPLY;
                        } else {
                            // Send REJECT (Type 4) or WAIT (Type 5)
                        }
                    }
                    break;

                case PE_SRC_TRANSITION_SUPPLY:
                    // Wait for PS_RDY (Control Message Type 6) after VBUS change
                    if (rx_msg_type == 0x06) { 
                        g_pd_state = PE_SRC_READY;
                    }
                    break;
                
                case PE_SRC_READY:
                    // Handle messages in steady state (e.g., Get_Sink_Cap)
                    if (rx_msg_type == 0x0A) { // PD_CTRL_GET_SINK_CAP
                        // TODO: Respond with Sink_Capabilities (if we were DRP)
                    }
                    break;
                
                default:
                    // Handle unexpected message in current state (e.g., request in READY state)
                    break;
            }
        } 
    }
    
    // C. Handle Hard Reset Received
    if (interrupt_a & INT_I_RXHRST) {
        // Clear Hard Reset bit (usually done by reading, but explicitly for clarity)
        // Set g_pd_state to initial state
        g_pd_state = PE_INIT; 
    }
    
    // D. Handle TX Failure/Retry Fail
    if (interrupt_a & (INT_I_RETRYFAIL)) {
        // FUSB302 attempts 3 retries automatically. If this fires, the message failed.
        // Initiate Hard Reset or retry negotiation.
        
        // Send a Hard Reset
        i2c_write_reg(FUSB302_REG_CONTROL3, CTL3_SEND_HARD_RESET); 
        g_pd_state = PE_INIT;
    }
}
